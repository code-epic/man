\section{Anexo A: Glosario de T\'erminos}

\item \textbf{TOTP}: Time-based One-Time Password
\item \textbf{RBAC}: Control de Acceso Basado en Roles
\item \textbf{gRPC}: Google Remote Procedure Call
\end{itemize}
